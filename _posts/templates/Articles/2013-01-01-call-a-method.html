Call a method

Dom events are handled by standard knockout event bindings. The context of the event handler will always be the current view model, even if it resides on another object

window.myType = wo.contentControl.extend(function myType() { this._super(); this.name = "My View Model"; }); myType.prototype.eventHandler1 = function() { alert(this.name); }; window.aStaticObject = { name: "My Static Object", eventHandler2: function() { alert(this.name); } }; <myType> <template> <button data-bind="click: eventHandler1">Test1</button> <button data-bind="click: aStaticObject.eventHandler2">Test2</button> </template> </myType> <wo.view templateId="'call-a-method-script-1'" />
