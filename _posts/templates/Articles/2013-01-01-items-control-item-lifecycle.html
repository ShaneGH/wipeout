List item lifecycle

With a custom itemsControl you can control what happens when list items are added and what happens when list items are removed.

var myItemsControl = wo.itemsControl.extend({ constructor: function() { this._super(); }, onItemRendered: function(item) { this._super(item); // do initialization logic here }, onItemDeleted: function(item) { this._super(item); // do cleanup logic here } });
